#include "FunctionBlockTypes.h"
#include "FBContainer.h"
#include "RuntimeEnvironment.h"

int main() {
    FBContainer container;

    auto block1 = std::make_shared<BasicFunctionBlock>();
    auto block2 = std::make_shared<BasicFunctionBlock>();

    block1->setValue("Block1 Initial Value");
    block2->setValue("Block2 Initial Value");

    container.addBlock("block1", block1);
    container.addBlock("block2", block2);

    container.setStartBlock("block1");

    RuntimeEnvironment runtime(container);
    runtime.process();

    return 1;
}
